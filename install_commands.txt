# Script para instalar dependências manualmente (Windows PowerShell)
# Execute cada comando abaixo no terminal, um por vez

# 1. Ativar o ambiente virtual (se já existe)
.venv\Scripts\Activate.ps1

# OU criar um novo ambiente virtual (se não existe)
# python -m venv .venv
# .venv\Scripts\Activate.ps1

# 2. Atualizar pip
pip install --upgrade pip

# 3. Instalar todas as dependências necessárias
pip install numpy
pip install opencv-python
pip install pillow
pip install scikit-learn
pip install matplotlib
pip install tensorflow

# 4. Verificar se tudo foi instalado corretamente
python -c "import numpy, cv2, PIL, sklearn, matplotlib, tensorflow; print('Todas as dependências instaladas com sucesso!')"

# 5. Para executar os scripts depois:
# python script.py
# python ia_treino.py
